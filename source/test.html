def build_network(net_path, summary_stats=False, plot=False):
    '''
    Create two networks from Friendly Ocean or Facebook ego network log
        1. friend-to-friend network
        2. ego network (#1 with EGO added)

    param(s):
        net_path - file path to either JSON log from Friendly Ocean or graphml file from Facebook scraper

    returns:
        the stats calcuated on the network
    '''

    pID=re.search('(DP\d+)(.json|_network)', net_path).group(1)

    print('Processing',pID)

    if net_path.endswith('.json'):


        net_json = json.load(open(net_path))

        NG = nx.Graph()

        # add nodes and associated data
        for friend in net_json['friends']:
            NG.add_node(friend['friendNumber'], friend)

        # add f2f links
        for link in net_json['links']:
            NG.add_edge(link['target'], link['source'])

    elif net_path.endswith('.graphml'):
        NG = nx.read_graphml(net_path)
        if len(NG.nodes())==0:
            return {'pID': pID}
    else:
        return

    # create EGO network by adding EGO with links to all friends
    NG_ego = NG.copy()
    for n in NG_ego.nodes():
        NG_ego.add_edge('EGO',n)

        if NG_ego.node[n].get('strength') is None:
            NG_ego.node[n]['strength']=4

    stats = {
        'pID': pID,
        'degree': len(NG.nodes()),
        'density': nx.density(NG),
        'ego_betweenness': nx.betweenness_centrality(NG_ego)['EGO'],
        'transitivity': nx.transitivity(NG_ego)
    }

    if summary_stats:
        for stat, value in stats.items():
            print(stat.title(), value)

    if plot:
        #nx.draw(NG)
        #plt.show()
        nx.draw(NG_ego,
                    node_color=['white' if n=='EGO' else '#B0B0FF' for n in NG_ego.nodes()],
                    node_size=[ 500 if n=='EGO' else NG_ego.node[n]['strength']*50 for n in NG_ego.nodes()]
               )

    return stats
